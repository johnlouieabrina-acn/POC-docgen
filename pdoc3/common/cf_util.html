""" Class : cf_util

    Purpose: It is a common util class that contains all the cloudformation template related code.
"""
import logging
import boto3
import botocore
import json


class CFUtil:
    """Common class for having utility functions for cloudformation."""
    def __init__(self, region_name):
        self.cloudformation_stack = boto3.client('cloudformation', region_name=region_name)

    def parse_template(self, template):
        """This function will parse the cloudformation template and look for any discrepancies.
            After that it will return the Body of the template.

            Args:
                template(template data)  : template data containing yaml content.

            Returns:
                template_data: Parsed template body contents.
        """
        template_data = ''
        try:
            template_data = template["Body"].read().decode()
        except TypeError as type_error:
            template_data = template.read()
            logging.error('Type error in the code %s' %type_error)
        except Exception as excep:
            template_data = template
            logging.error('Exception in the code %s' %excep)
        finally:
            self.cloudformation_stack.validate_template(TemplateBody=template_data)
        return template_data

    def stack_exists(self, stack_name):
        """This method checks if stack is already existing or getting processed.

            Args:
                stack_name(stack_name)  : Stack name containing the cloudformation stack.

            Returns:
                status : True if it is running, False if not running.
        """
        paginator = self.cloudformation_stack.get_paginator('list_stacks')
        response_iterator = paginator.paginate(StackStatusFilter=['CREATE_COMPLETE', 'UPDATE_COMPLETE'
                                                                    'UPDATE_ROLLBACK_COMPLETE'])
        for page in response_iterator:
            stack = page['StackSummaries']
            for output in stack:
                if output['StackStatus'] == 'DELETE_COMPLETE':
                    continue
                if stack_name == output['StackName']:
                    return True
        return False

    def create_resource(self, stack_name, template, parameter_local, role):
        """This function will create the resource based on cloudformation and other details.

            Args:
                stack_name(string) : stack name of the cloudformation.
                template(template data)  : template data containing yaml content.
                parameter_local (map) : parameter data containing inputs needed for cloudformation.
                role(string) : iam role needed for creating.
                region_name(string) : region name for the stack.

            Returns:
                None if success, else it will throw error.
        """
        template_data = self.parse_template(template)
        Capability = ['CAPABILITY_NAMED_IAM']

        #    'RoleARN': role, can be commented out from above 'parms' if setting up user has previleges
        params = {
            'StackName': stack_name,
            'TemplateBody': template_data,
            'Parameters': parameter_local,
            'RoleARN': role,
            'Capabilities': Capability
        }

        logging.info('\tFetched Environment Variables\n\t______________________________')
        for par in parameter_local:
            logging.info('\t{} :\t {}'.format(par['ParameterKey'], par['ParameterValue']))
        logging.info('\n')

        try:
            if self.stack_exists(stack_name):
                logging.info('Updating {}'.format(stack_name))
                stack_result = self.cloudformation_stack.update_stack(**params)
                waiter = self.cloudformation_stack.get_waiter('stack_update_complete')
            else:
                logging.info('Creating {}'.format(stack_name))
                stack_result = self.cloudformation_stack.create_stack(**params)
                waiter = self.cloudformation_stack.get_waiter('stack_create_complete')
            logging.info("...waiting for stack to be ready...")
            waiter.wait(StackName=stack_name)
        except botocore.exceptions.ClientError as ex:
            error_message = ex.response['Error']['Message']
            if error_message == 'No updates are to be performed.':
                logging.info("~~~~~~~~~~~~~No changes~~~~~~~~~~~~~")
            else:
                raise
        else:
            stackid = stack_result['StackId']
            logging.info('Stack Is Deployed!'.center(80, '-'))
            result = "STACK ID : {}".format(stackid)
            logging.info(result)

    def delete_resource(self,stack_name):
        """This function will delete the resource based on cloudformation and other details.

            Args:
                stack_name(string) : stack name of the cloudformation.

            Returns:
                None if success, else it will throw error.
        """
        try:
            response = self.cloudformation_stack.delete_stack(
                    StackName=stack_name,
                    )
            # Check success response
            if 'ResponseMetadata' in response and \
                    response['ResponseMetadata']['HTTPStatusCode'] < 300:
                logging.info ("stack deleted %s "%(stack_name))
                logging.info("Response is {0}".format(json.dumps(response)))
                waiter = self.cloudformation_stack.get_waiter('stack_delete_complete')
                try:
                    # Waits for the stack to get completed. This will check
                    # every 15 seconds, for at most 5 minutes (60 / 15 * 5 = 20 attempts)
                    waiter.wait(StackName=stack_name,
                                WaiterConfig={
                                    'Delay': 15,
                                    'MaxAttemps': 20
                                })
                    logging.info(' Waiting for stack to get deleted %s '%(stack_name))
                except:
                    logging.info(' stack not deleted %s '%(stack_name))
            else:
                logging.error("There was an Unexpected error. response: {0}".format(json.dumps(response)))
        except TypeError as type_error:
            logging.error('Type error in the code %s' %type_error)
        except botocore.exceptions.ClientError as ex:
            logging.error("Boto3 error in create_resource method of cf_util %s" %(ex))
